\begin{frame}{}
  \begin{center}
    假设有向图中节点编号为0\purple{(初始状态)}到 $n - 1$

    \vspace{0.50cm}
    $R^{k}_{ij}:$ 从节点 $i$ 到节点 $j$、且\red{\bf 中间节点}编号\red{\bf 不大于$k$}的所有路径

    \pause
    \fig{width = 0.50\textwidth}{figs/dfa-re-kleene}
    \[
      R^{k}_{ij} = R_{ik}^{k-1} \blue{(R^{k-1}_{kk})^{\ast}} R^{k-1}_{kj}
        \mid R_{ij}^{k-1}
    \]
  \end{center}
\end{frame}
