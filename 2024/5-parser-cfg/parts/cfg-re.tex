\begin{frame}{}
  \begin{theorem}
    $L = \set{a^{n} b^{n} \mid n \ge 0}$ 无法使用正则表达式描述。
  \end{theorem}

  \pause
  \begin{center}
    \red{\bf 反证法}

    \pause
    \vspace{0.30cm}
    假设存在正则表达式 $r$: $L(r) = L$

    \pause
    \vspace{0.30cm}
    则存在\blue{\bf 有限}状态自动机 $D(r)$: $L(D(r)) = L$; 设其状态数为 $k$

    \pause
    \vspace{0.30cm}
    \fbox{\red{考虑输入 $a^{m}\; (m > k)$}}
    \fig{width = 0.90\textwidth}{figs/pumping-lemma-anbn}

    \pause
    \vspace{0.30cm}
    $D(r)$ 也能接受 $a^{i+j} b^{i}$; \red{\bf 矛盾!}
  \end{center}
\end{frame}
