\begin{algorithm}[H]
% \caption{}
% \label{alg:S}
\begin{algorithmic}[1]
  \Procedure{\blue{$S$}}{\null}
    \If{\red{token = `('}}
      \State \Call{match}{`('}
      \State \Call{$S$}{\null}
      \State \Call{match}{`+'}
      \State \Call{$F$}{\null}
      \State \Call{match}{`)'}
    \ElsIf{\red{token = `a'}}
      \State \Call{$F$}{\null}
    \Else
      \State \Call{\teal{error}}{token, \set{`(', `a'}}
    \EndIf
  \EndProcedure
\end{algorithmic}
\end{algorithm}
