\begin{algorithm}[H]
% \caption{}
% \label{alg:S}
\begin{algorithmic}[1]
  \Procedure{match}{$t$} % \Comment{$t$ 是预期的词法单元}
    \If{token = $t$}  % \Comment{token 是输入中当前的词法单元}
      \State token $\gets \Call{\purple{next-token}}{\null}$
    \Else
      \State \Call{error}{token, $t$}
    \EndIf
  \EndProcedure
\end{algorithmic}
\end{algorithm}
