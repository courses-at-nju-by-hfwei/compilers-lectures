  \end{columns}

  \vspace{0.50cm}
	{\small
  \begin{itemize}[<+->]
	\setlength{\itemsep}{8pt}
	\item \red{Launch subparsers at a decision point, one per alternative productions.}
	\item \teal{These subparsers run in pseudo-parallel to explore all possible paths.}
	\vspace{10pt}
	\item \blue{Subparsers die off as their paths fail to match the remaining input.}
	\vspace{10pt}
	\item \purple{Ambiguity: Multiple subparsers coalesce together or reach EOF.}
	\item \purple{Resolution: The first production associated with a surviving subparser.}
  \end{itemize}}
\end{frame}
