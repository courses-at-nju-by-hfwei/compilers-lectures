\begin{frame}{}
  \begin{definition}[字符串 $x$ 区分状态 $s$ 与 $t$]
	如果分别从 $s$ 与 $t$ 出发, 沿着标号为 $x$ 的路径到达的两个状态中
	只有一个是接受状态, 则称 $x$ \red{\bf 区分}了状态 $s$ 与 $t$。
  \end{definition}

  \pause
  \vspace{1.00cm}
  \begin{definition}[可区分的 (Distinguishable)]
	如果存在某个能区分状态 $s$ 与 $t$ 的字符串,
	则称 $s$ 与 $t$ 是\red{\bf 可区分的}。
  \end{definition}
\end{frame}
